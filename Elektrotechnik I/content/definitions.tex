\paragraph{Electric Charge:} % (fold)
	$
		e = \SI{1.6022e-19}{\coulomb}
	$
	\vspace{-2mm}
	\begin{tightitemize}
		\item can be positive or negative
		\item is quantized
		\item is conserved
	\end{tightitemize}
% paragraph: Electric Charge (end)

\paragraph{Coulomb's Law:} % (fold)
	$\displaystyle
		F = k \frac{q_1 q_2}{r^2} \\
		k = \frac{1}{4 \pi \varepsilon_0} \qquad
		\varepsilon_0 = \SI{8.85e-12}{\Square\coulomb\per\newton\per\Square\metre}
	$
% paragraph: Coulomb's Law (end)

\paragraph{Electric Field:} % (fold)
	$\displaystyle
		E = \frac{F}{q_0} \sunit{\volt\per\meter}
	$
% paragraph: Electric Field (end)

\paragraph{Electric Potential:} % (fold)
	$\displaystyle
		V = \frac{U}{q_0} \sunit{\volt} = \unit{\joule\per\coulomb} \\
		U_a - U_b = - \int_a^b F \diff s
	$
% paragraph: Electric Potential (end)

\paragraph{Current:} % (fold)
	$\displaystyle
		I = \lim_{\Delta t \to 0} \frac{\Delta Q}{\Delta t} = \frac{\partial Q}{\partial t} \sunit{A} = \unit{\coulomb\per\second} \\
		Q = I \cdot t \sunit{C}
	$
% paragraph: Current (end)

\paragraph{Joule's Law:} % (fold)
	$\displaystyle
		P = \frac{\partial U}{\partial t}
		  = I \, V
		  = I^2 \, R
		  = \frac{V^2}{R}
		\sunit{W} = \unit{\joule\per\second}
	$
% paragraph: Joule's Law (end)

\paragraph{Ohm's Law:} % (fold)
	$\displaystyle
		I = \nicefrac{V}{R}
	$
% paragraph: Ohm's Law (end)
